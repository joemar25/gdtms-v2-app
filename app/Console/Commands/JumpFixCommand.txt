<?php

namespace App\Console\Commands;

use Illuminate\Console\Command;
use Symfony\Component\Process\Process;

class JumpFixCommand extends Command
{
    /**
     * The name and signature of the console command.
     *
     * @var string
     */
    protected $signature = 'native:jump-fixed {platform=android} {--skip-build}';

    /**
     * The console command description.
     *
     * @var string
     */
    protected $description = 'Start the NativePHP Jump server using relative paths to avoid Windows encoding issues';

    /**
     * Execute the console command.
     */
    public function handle()
    {
        $platform = $this->argument('platform');

        $this->components->info("Starting Fixed Jump Server for {$platform}");

        // 1. Detect Network IP
        $displayHost = $this->getNetworkIp();
        $httpPort = 8001;
        $laravelPort = 8000;

        // 2. Prepare Paths
        $zipPath = base_path('storage/app/native-build/app.zip');
        $routerPath = base_path('resources/native/jump-router.php');

        // Check if zip exists
        if (!file_exists($zipPath)) {
            $this->error("App bundle not found at: {$zipPath}");
            $this->info("Please run 'php artisan native:build {$platform}' first.");
            return Command::FAILURE;
        }

        if (!file_exists($routerPath)) {
            $this->error("Jump router not found at: {$routerPath}");
            return Command::FAILURE;
        }

        // Use custom router with improved device detection (bypasses 'ñ' path bug)
        $relativeRouterPath = 'resources/native/jump-router.php';

        $this->line("  Display Host: <info>{$displayHost}</info>");
        $this->line("  Jump Port:    <info>{$httpPort}</info>");
        $this->line("  Laravel Port: <info>{$laravelPort}</info>");
        $this->newLine();

        // 3. Start the Laravel dev server on port 8000 in the background
        //    The jump router proxies all app requests to this server.
        $this->components->info("Starting Laravel server on port {$laravelPort}...");

        $laravelProcess = new Process(
            [PHP_BINARY, 'artisan', 'serve', '--port=' . $laravelPort, '--host=127.0.0.1'],
            getcwd(),
            null,
            null,
            null  // no timeout
        );

        $laravelProcess->start(function ($type, $buffer) {
            // Discard output — Laravel logs to storage/logs/laravel.log
        });

        // Wait up to 3 seconds for the Laravel server to become ready
        $ready = false;
        $deadline = time() + 3;
        while (time() < $deadline) {
            usleep(300000); // 300ms
            $ch = curl_init("http://127.0.0.1:{$laravelPort}/");
            curl_setopt($ch, CURLOPT_RETURNTRANSFER, true);
            curl_setopt($ch, CURLOPT_CONNECTTIMEOUT, 1);
            curl_setopt($ch, CURLOPT_TIMEOUT, 1);
            curl_exec($ch);
            $errno = curl_errno($ch);
            curl_close($ch);

            // Connection refused (7) means not up yet; any other response means it's up
            if ($errno !== CURLE_COULDNT_CONNECT) {
                $ready = true;
                break;
            }
        }

        if (!$ready) {
            $this->warn("Laravel server did not respond in time — continuing anyway.");
        } else {
            $this->components->info("Laravel server ready on port {$laravelPort}.");
        }

        // Ensure the Laravel process is stopped when this script exits
        register_shutdown_function(function () use ($laravelProcess) {
            if ($laravelProcess->isRunning()) {
                $laravelProcess->stop(3);
            }
        });

        $this->newLine();
        $this->components->info("QR Link: http://{$displayHost}:{$httpPort}/jump/qr");
        $this->newLine();
        $this->info("Servers running... Press Ctrl+C to stop.");

        // 4. Set env variables for the current process so passthru inherits them
        $env = [
            'JUMP_ZIP_PATH' => 'storage/app/native-build/app.zip',
            'JUMP_DISPLAY_HOST' => $displayHost,
            'JUMP_HTTP_PORT' => (string) $httpPort,
            'JUMP_LARAVEL_PORT' => (string) $laravelPort,
            'JUMP_BASE_PATH' => '.',
            'APP_NAME' => config('app.name', 'Laravel'),
        ];

        foreach ($env as $key => $value) {
            putenv("{$key}={$value}");
        }

        // 5. Start the jump server (blocks until Ctrl+C)
        $jumpCommand = sprintf(
            'php -S 0.0.0.0:%d %s',
            $httpPort,
            $relativeRouterPath
        );

        passthru($jumpCommand);

        return Command::SUCCESS;
    }

    /**
     * Get the first non-loopback IPv4 address
     */
    protected function getNetworkIp(): string
    {
        $ips = [];
        exec('ipconfig', $output);

        foreach ($output as $line) {
            if (preg_match('/IPv4 Address.*: ([\d\.]+)/', $line, $matches)) {
                if (!str_starts_with($matches[1], '127.')) {
                    $ips[] = $matches[1];
                }
            }
        }

        return !empty($ips) ? $ips[0] : '127.0.0.1';
    }
}
